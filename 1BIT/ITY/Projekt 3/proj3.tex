\SetNlSty{}{}{:}
\SetNlSkip{-1mm}
\IncMargin{1em}
\begin{algorithm}[H]
\caption{\textsc{FastSLAM}\label{algoritmus}}
\Indm
\KwIn{$(X_{t -1}, u_t, z_t)$}
\KwOut{$X_t$}
\BlankLine
\Indp\Indpp
$\overline{X_t} = X_t = 0$\\
\For{$k = 1 $\ {\upshape to}\ $ M$}{$x_{t}^{[k]} = sample\_motion\_model(u_t,x_{t-1}^{[k]})$\par
$\omega_t^{[k]} = measurement\_model(z_t,x_t^{[k]},m_{t-1})$\par
$m_t^{[k]} = updated\_occupancy\_grid(z_t,x_t^{[k]},m_{t-1}^{[k]})$\par
$\overline{X_t} = \overline{X_t} + \langle x_x^{[m]},\omega_t^{[m]}\rangle$}
\For{$k = 1$\ {\upshape to} $M$}{draw $i$ with probability $\approx \omega_t^{[i]}$\par
add $\langle x_x^{[k]},m_t^{[k]}\rangle$ to $X_t$}
\KwRet{$X_t$}
\end{algorithm}
